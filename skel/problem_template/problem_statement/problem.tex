\problemname{Identity}

\newcommand{\maxn}{10^5}

You are required to print the number given in the input.

\begin{Input}
	\begin{itemize}
		\item The input consists of one line, containing a single integer $0\leq n\leq \maxn$.
	\end{itemize}
\end{Input}

\begin{Output}
	\begin{itemize}
		\item The output consists of a single line containing the number $n$.
	\end{itemize}
\end{Output}

\begin{Sample}
	0 & 0 \\
\end{Sample}

\begin{Sample}
	% Use \newline after every line if the input or output consists of multiple lines.
	42 %\newline
	&
	42 %\newline
	\\
\end{Sample}
